## 오큰수 
```python
# 시간초과

N = int(input())
n = list(map(int, input().split()))
answer = []
stack = []
max_m = -1

for i in range(len(n)):
    stack.append(n[i])
    if i == 0 or n[i] > stack[i-1]:
        check = 0
        point = i + 1
        while point < len(n):
            if n[point] > stack[i]:
                answer.append(n[point])
                max_m = n[point]
                check = 1
                break
            else:
                point += 1
        if check == 0:
            answer.append(-1)
    elif i == len(n) - 1:
        answer.append(-1)
    elif n[i] <= stack[-1]:
        answer.append(max_m)

        
print(*answer, sep=' ')

```

```python
# 코드를 더 간단하게 풀어봤는데 이것도 시간초과

N = int(input())
n = list(map(int, input().split()))
answer = []

for i in range(len(n)):
    stack = n[i:]
    m = stack.pop(0)
    if not stack or m >= max(stack):
        answer.append(-1)
    elif m < stack[0]:
        answer.append(stack[0])
    else:
        answer.append(max(stack))
        
print(*answer, sep=' ')

```

```python
# 시간초과 안나는 답

n = int(input())
A = list(map(int, input().split()))
answer = [-1]*n
stack = []

stack.append(0)
for i in range(1, n):
    while stack and A[stack[-1]] < A[i]:
        answer[stack.pop()] = A[i]
    stack.append(i)

print(*answer)

```

## 오등큰수 
```python
# 시간초과

import sys

n = int(input())
a = list(map(int, sys.stdin.readline().split()))
b = [a.count(i) for i in a]
answer = [-1]*n
stack = [0]

for i in range(1, n):
    while stack and b[stack[-1]] < b[i]:
        answer[stack.pop()] = a[i]
    stack.append(i)

print(*answer)    

```

```python
# 시간초과 안나는 답. 거의 비슷한데 여기서는 Counter를 썼다. 역시 해쉬 빠름.

from collections import Counter
import sys

n = int(input())
a = list(map(int, sys.stdin.readline().split()))
b = Counter(a)
answer = [-1]*n
stack = [0]

for i in range(1, n):
    while stack and b[a[stack[-1]]] < b[a[i]]:
        answer[stack.pop()] = a[i]
    stack.append(i)

print(*answer)    

```
