# 맨 처음 답 시간초과

import sys


n, s = map(int, input().split())
place = list(map(int, sys.stdin.readline().split()))
sib = []

for i in range(n):
    sib.append(abs(place[i] - s))

def div(n):
    data = []
    for i in range(1, n//2+1):
        if n % i == 0:
            data.append(i)
    data.append(n)
    return list(reversed(data))

l_div = div(min(sib))
for i in l_div:
    check = 0
    for j in sib:
        if j % i != 0:
            check = 1
            break
        else:
            continue
    if check == 0:
        print(i)
        break    

```

# 하루 자고 머리 비우고 다시 풀어 맞춘 답.. 박수드립니다.

import sys

n, s = map(int, input().split())
sib_n = list(map(int, sys.stdin.readline().split()))
sib = [abs(i-s) for i in sib_n]

def gcd(a, b):
    while b > 0:
        a, b = b, a % b
    return a

if n == 1:
    print(*sib)
elif n == 2:
    print(gcd(sib[0], sib[1]))
else:
    G = gcd(sib[0], sib[1])
    for i in range(len(sib[2:])):
        G = gcd(G, sib[i])
    print(G)
    
```
